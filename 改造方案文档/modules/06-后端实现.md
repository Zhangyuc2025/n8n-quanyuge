  /**
   * âœ… æäº¤èŠ‚ç‚¹å®¡æ ¸ï¼ˆå¯é€‰ï¼‰
   */
  async submitForReview(nodeId: string, workspaceId: string) {
    const node = await this.customNodeRepository.findOneOrFail({
      where: { id: nodeId, workspaceId },
    });

    if (node.submissionStatus && node.submissionStatus !== 'draft') {
      throw new UserError('Node has already been submitted');
    }

    node.submissionStatus = 'pending';
    node.submittedAt = new Date();

    return await this.customNodeRepository.save(node);
  }

  /**
   * âœ… ç®¡ç†å‘˜å®¡æ ¸èŠ‚ç‚¹
   */
  async reviewNode(
    nodeId: string,
    reviewerId: string,
    approved: boolean,
    reviewNotes?: string
  ) {
    const node = await this.customNodeRepository.findOneOrFail({
      where: { id: nodeId },
    });

    node.submissionStatus = approved ? 'approved' : 'rejected';
    node.reviewedBy = reviewerId;
    node.reviewedAt = new Date();
    node.reviewNotes = reviewNotes || null;

    if (approved) {
      // âœ… å®¡æ ¸é€šè¿‡ï¼šè½¬ç§»åˆ° platform_node è¡¨
      await this.convertToPlatformNode(node);

    return await this.customNodeRepository.save(node);

      nodeKey: `${customNode.workspaceId}-${customNode.nodeKey}`,
      nodeName: customNode.nodeName,
      nodeType: 'third_party_approved',
      category: customNode.category,
      description: customNode.description,
      iconUrl: customNode.iconUrl,
      version: customNode.version,
      nodeDefinition: customNode.nodeDefinition,
      nodeCode: customNode.nodeCode,
      submissionStatus: 'approved',
      submittedBy: customNode.createdBy,
      submittedAt: customNode.submittedAt,
      reviewedBy: customNode.reviewedBy,
      reviewedAt: customNode.reviewedAt,
      reviewNotes: customNode.reviewNotes,
      isActive: true,
      enabled: true,
    });

    await this.platformNodeRepository.save(platformNode);
  }

@Service()
export class PlatformAIService {
  constructor(
    private readonly platformAIProviderRepository: PlatformAIProviderRepository,
    private readonly billingService: BillingService,
    private readonly cipher: Cipher,
  ) {}

  /**
   * è·å–æä¾›å•†é…ç½®
   */
  async getProvider(providerKey: string) {
    const provider = await this.platformAIProviderRepository.findOne({
      where: { providerKey, isActive: true, isEnabled: true },
    });

    if (!provider) {
      throw new NotFoundError(`AI provider '${providerKey}' not found`);
    }

    return provider;
  }

  /**
   * è·å–æ¨¡å‹åˆ—è¡¨
   */
  async getModels(providerKey: string) {
    const provider = await this.getProvider(providerKey);
    return provider.modelsConfig;
  }

  /**
   * æ‰§è¡ŒèŠå¤©ï¼ˆå¸¦è®¡è´¹ï¼‰
   */
  async chat(
    providerKey: string,
    workspaceId: string,
    userId: string,
    params: {
      model: string;
      messages: Array<{ role: string; content: string }>;
      temperature?: number;
      maxTokens?: number;
    }
  ) {
    const provider = await this.getProvider(providerKey);

    // è§£å¯† API é…ç½®
    const apiConfig = JSON.parse(this.cipher.decrypt(provider.apiConfig));

    // è°ƒç”¨ä¸Šæ¸¸ API
    const response = await this.callUpstreamAPI(providerKey, apiConfig, params);

    // è®¡ç®—è´¹ç”¨
    const tokensUsed = response.usage.total_tokens;
    const model = provider.modelsConfig.find(m => m.id === params.model);
    const cost = (tokensUsed / 1000) * model.pricePerToken;

    // æ‰£è´¹
    await this.billingService.charge({
      workspaceId,
      userId,
      serviceKey: `${providerKey}:${params.model}`,
      serviceType: 'ai_model',
      tokensUsed,
      cost,
      metadata: {
        model: params.model,
        messagesCount: params.messages.length,
      },
    });

    return response;
  }

  /**
   * è°ƒç”¨ä¸Šæ¸¸ API
   */
  private async callUpstreamAPI(
    providerKey: string,
    apiConfig: any,
    params: any
  ): Promise<any> {
    switch (providerKey) {
      case 'openai':
        return await this.callOpenAI(apiConfig, params);
      case 'anthropic':
        return await this.callAnthropic(apiConfig, params);
      default:
        throw new Error(`Unsupported provider: ${providerKey}`);
    }
  }

  private async callOpenAI(apiConfig: any, params: any) {
    const response = await fetch(`${apiConfig.baseUrl}/chat/completions`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${apiConfig.apiKey}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: params.model,
        messages: params.messages,
        temperature: params.temperature ?? 0.7,
        max_tokens: params.maxTokens,
      }),
    });

    if (!response.ok) {
      throw new Error(`OpenAI API error: ${response.statusText}`);
    }

    return await response.json();
  }

  private async callAnthropic(apiConfig: any, params: any) {
    // Anthropic API çš„å®ç°
    const response = await fetch(`${apiConfig.baseUrl}/messages`, {
      method: 'POST',
      headers: {
        'x-api-key': apiConfig.apiKey,
        'anthropic-version': '2023-06-01',
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        model: params.model,
        messages: params.messages,
        temperature: params.temperature ?? 0.7,
        max_tokens: params.maxTokens ?? 1024,
      }),
    });

    if (!response.ok) {
      throw new Error(`Anthropic API error: ${response.statusText}`);
    }

    const data = await response.json();

    // è½¬æ¢ä¸ºç»Ÿä¸€æ ¼å¼
    return {
      choices: [
        {
          message: {
            role: 'assistant',
            content: data.content[0].text,
          },
        },
      ],
      usage: {
        prompt_tokens: data.usage.input_tokens,
        completion_tokens: data.usage.output_tokens,
        total_tokens: data.usage.input_tokens + data.usage.output_tokens,
      },
    };
  }
}
